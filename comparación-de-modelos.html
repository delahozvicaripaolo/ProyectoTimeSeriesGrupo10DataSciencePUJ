comparar_modelos <- function(nombre_indice) {
  # Validar que existan resultados
  if (is.null(resultados_arima[[nombre_indice]]) ||
      is.null(resultados_hw[[nombre_indice]]) ||
      is.null(resultados_prophet[[nombre_indice]])) {
    cat("⚠️  No se encontraron resultados completos para", nombre_indice, "\n")
    return(NULL)
  }
  
  serie_obs <- cierres_zoo[[nombre_indice]]
  df_obs <- data.frame(
    Fecha = index(serie_obs),
    Valor = as.numeric(serie_obs),
    Tipo = "Observado"
  )

  # ARIMA
  forecast_arima <- resultados_arima[[nombre_indice]]$pronostico
  if (is.null(forecast_arima$mean)) {
    cat("❌ ARIMA falló en", nombre_indice, "\n")
    return(NULL)
  }
  df_arima <- data.frame(
    Fecha = time(forecast_arima$mean),
    Valor = as.numeric(forecast_arima$mean),
    Modelo = "ARIMA"
  )

  # Holt-Winters
  forecast_hw <- resultados_hw[[nombre_indice]]$pronostico
  if (is.null(forecast_hw$mean)) {
    cat("❌ Holt-Winters falló en", nombre_indice, "\n")
    return(NULL)
  }
  df_hw <- data.frame(
    Fecha = time(forecast_hw$mean),
    Valor = as.numeric(forecast_hw$mean),
    Modelo = "Holt-Winters"
  )

  # Prophet
  forecast_prophet <- resultados_prophet[[nombre_indice]]$forecast
  df_prophet <- forecast_prophet %>%
    select(ds, yhat) %>%
    rename(Fecha = ds, Valor = yhat) %>%
    mutate(Modelo = "Prophet") %>%
    filter(Fecha > max(df_obs$Fecha))

  df_obs$modelo <- "Observado"
  df_total <- bind_rows(df_obs %>% rename(Modelo = Tipo), df_arima, df_hw, df_prophet)

  ggplot(df_total, aes(x = Fecha, y = Valor, color = Modelo)) +
    geom_line(size = 0.8) +
    labs(
      title = paste("Comparación de modelos -", nombre_indice),
      x = "Fecha",
      y = "Valor",
      color = "Modelo"
    ) +
    theme_minimal() +
    scale_color_manual(values = c("black", "red", "green", "blue")) +
    theme(plot.title = element_text(face = "bold", size = 14))
}
