8.1 Implementación

Se utiliza RSNNS para las redes recurrentes clásicas (Elman y Jordan), Metrics para errores, ggplot2 para visualización, dplyr / tibble para manipulación de tablas y zoo para las series temporales de cierres diarios.

El particionamiento se realizó de la siguiente forma: 70% entrenamiento, 15% validación temprana y 15% prueba. Para el escalado de utilizó Z-score.

Se realizó la validación de hiperparametros de forma independiente con el resultado que la mejor combinación es:

Size (Numero de neuronas entrenadas): 25. lr (Learning rate):0.01 Maxix(Épocas máximas):400.

Se implementó un Lag de 12 observaciones y para las métricas de medición se utilizáron las siguientes: MAE, RMSE y MAPE.

calc_logret <- function(z) diff(log(z)) |> na.omit()

## 1.1  dummies de día (lunes referencia)
dow_dum <- function(idx){
  lev <- c("Tuesday","Wednesday","Thursday","Friday")
  outer(weekdays(idx), lev, "==") * 1          # matriz  n × 4
}

## 1.2  Ventanas:  y(t)  + 12 lags  + 4 dummies  (16 inputs)
crear_ventanas <- function(r, n_lag = 12){
  emb_r <- embed(coredata(r), n_lag + 1)
  idx   <- time(r)[(n_lag + 1):length(r)]      # fechas alineadas con y(t)
  Xret  <- emb_r[, 2:(n_lag + 1)]
  Xdow  <- dow_dum(idx)
  df    <- as.data.frame(cbind(y = emb_r[,1], Xret, Xdow))
  colnames(df) <- c("y",
                    paste0("lag", 1:n_lag),
                    c("Tue","Wed","Thu","Fri"))
  df                                           # cronológico asc
}

particionar <- function(df, p_train=.70, p_val=.15){
  n <- nrow(df); nt <- floor(n*p_train); nv <- floor(n*p_val)
  list(train=df[1:nt,], val=df[(nt+1):(nt+nv),], test=df[(nt+nv+1):n,])
}

prox_habiles <- function(fecha_ini, n){
  s <- seq(fecha_ini + 1, by = "day", length.out = n*3)
  s[!weekdays(s) %in% c("Saturday","Sunday")][1:n]
}

## 1.3  Entrenamiento RNN
entrenar_rnn <- function(parts, tipo=c("elman","jordan"),
                         size=25, lr=0.01, maxit=400){
  tipo <- match.arg(tipo)

  μ <- mean(parts$train$y); parts <- lapply(parts,\(d){d$y<-d$y-μ;d})
  σ <- sd(parts$train$y); if (σ==0) σ <- 1e-6
  sc  <- \(v) v/σ;  isc <- \(v) v*σ

  toMat <- \(d) as.matrix(d[ , -1])
  trX <- sc(toMat(parts$train)); trY <- sc(parts$train$y)
  vaX <- sc(toMat(parts$val  )); vaY <- sc(parts$val$y)
  teX <- sc(toMat(parts$test )); teY <- parts$test$y

  net <- if (tipo=="elman"){
           elman (trX,trY,size,linOut=TRUE,
                  learnFuncParams=c(lr),
                  inputsTest=vaX,targetsTest=vaY,maxit=maxit)
         } else {
           jordan(trX,trY,size,linOut=TRUE,
                  learnFuncParams=c(lr),
                  inputsTest=vaX,targetsTest=vaY,maxit=maxit)
         }

  pred_te <- isc(predict(net, teX))
  list(net=net, sc=sc, isc=isc,
       clip = 2*sd(parts$train$y),           # límite de clipping por índice
       MAE = mae (teY, pred_te),
       RMSE= rmse(teY, pred_te),
       MAPE= mape(teY, pred_te)*100)
}

plot_fc <- function(idx, fc_price, modelo){
  obs    <- cierres_zoo[[idx]]
  fechas <- prox_habiles(tail(index(obs),1), length(fc_price))
  bind_rows(
      tibble(Fecha=index(obs), Valor=as.numeric(obs), Serie="Observado"),
      tibble(Fecha=fechas    , Valor=fc_price      , Serie="Pronóstico")) |>
    ggplot(aes(Fecha,Valor,col=Serie,linetype=Serie))+
      geom_line(size=.9)+
      scale_color_manual(values=c(Observado="#1f77b4",Pronóstico="#d62728"))+
      labs(title=sprintf("Pronóstico 30 días – %s (%s)", idx, modelo),
           y="Precio de cierre")+theme_minimal()
}

logret_zoo <- lapply(cierres_zoo, \(z){ r<-calc_logret(z); r-mean(r) })
names(logret_zoo) <- nombres_indices

#───────────────────────────────────────────────
# 3.  Loop entrenamiento / forecast
#───────────────────────────────────────────────
metricas_elman  <- tibble(); resultados_elman  <- list()
metricas_jordan <- tibble(); resultados_jordan <- list()

for(idx in nombres_indices){
  r <- logret_zoo[[idx]]
  if(length(r) < 260){ message(idx,": pocos datos"); next }

  df_lags <- crear_ventanas(r, 12)
  parts   <- particionar(df_lags, .70, .15)

  ## ----------  ELMAN  ----------
  fitE <- entrenar_rnn(parts,"elman",25,.01,400)

  # pronóstico paso a paso (30 días hábiles)
  last_price <- as.numeric(tail(cierres_zoo[[idx]],1))
  ret_fc <- numeric(30); win <- tail(df_lags$y,12)
  next_dates <- prox_habiles(time(tail(cierres_zoo[[idx]],1)), 30)
  for(h in seq_len(30)){
    x <- c(fitE$sc(win), dow_dum(next_dates[h]))
    r_hat <- fitE$isc(predict(fitE$net, matrix(x,1)))
    ret_fc[h] <- pmin(pmax(r_hat, -fitE$clip), fitE$clip)
    win <- c(ret_fc[h], head(win,-1))
  }
  resultados_elman[[idx]] <- last_price*exp(cumsum(ret_fc))

  metricas_elman <- bind_rows(metricas_elman,
      tibble(Indice=idx, MAE=round(fitE$MAE,4),
             RMSE=round(fitE$RMSE,4), MAPE=round(fitE$MAPE,2)))

  ## ----------  JORDAN ----------
  fitJ <- entrenar_rnn(parts,"jordan",25,.01,400)

  ret_fc <- numeric(30); win <- tail(df_lags$y,12)
  for(h in seq_len(30)){
    x <- c(fitJ$sc(win), dow_dum(next_dates[h]))
    r_hat <- fitJ$isc(predict(fitJ$net, matrix(x,1)))
    ret_fc[h] <- pmin(pmax(r_hat, -fitJ$clip), fitJ$clip)
    win <- c(ret_fc[h], head(win,-1))
  }
  resultados_jordan[[idx]] <- last_price*exp(cumsum(ret_fc))

  metricas_jordan <- bind_rows(metricas_jordan,
      tibble(Indice=idx, MAE=round(fitJ$MAE,4),
             RMSE=round(fitJ$RMSE,4), MAPE=round(fitJ$MAPE,2)))
}
